精细提示词 vs 粗略提示词生成的 Vue 组件真实差异

❌ 粗略版本 - 代码 
import { onMounted } from 'vue'
import L from 'leaflet'
import 'leaflet/dist/leaflet.css'

// 问题1：没有类型定义
// 问题2：没有 Props 配置
// 问题3：没有清理逻辑

onMounted(() => {
  // 问题4：使用ID选择器，可能冲突
  const map = L.map('map-container')
  
  L.tileLayer('https://...').addTo(map)
  
  // 问题5：硬编码坐标
  L.marker([39.9042, 116.3974]).addTo(map)
  
  // 问题6：没有保存实例，无法清理
})
✅ 精细版本 - 代码 
import { onMounted, onUnmounted, shallowRef, ref, withDefaults } from 'vue'
import L from 'leaflet'
import 'leaflet/dist/leaflet.css'

// ✅ 优势1：完整类型定义
interface Props {
  center?: [number, number]
  zoom?: number
  tileUrl?: string
}

const props = withDefaults(defineProps<Props>(), {
  // ✅ 优势2：可配置默认值
  center: () => [39.9042, 116.3974] as [number, number],
  zoom: 13,
  tileUrl: 'https://...'
})

const mapContainer = ref<HTMLElement | null>(null)
const mapInstance = shallowRef<L.Map | null>(null)

onMounted(() => {
  // ✅ 优势3：使用 ref 避免ID冲突
  if (!mapContainer.value) return
  
  const map = L.map(mapContainer.value, {
    center: props.center,
    zoom: props.zoom
  })
  
  L.tileLayer(props.tileUrl, { maxZoom: 19 }).addTo(map)
  L.marker(props.center).addTo(map)
  
  mapInstance.value = map
  setTimeout(() => map.invalidateSize(), 200)
})

// ✅ 优势4：完整清理生命周期
onUnmounted(() => {
  mapInstance.value?.remove()
  mapInstance.value = null
})

// ✅ 优势5：暴露方法给父组件
defineExpose({
  getMapInstance: () => mapInstance.value
})

🔍 5大核心差异详解

1️⃣ TypeScript 类型安全

精细版本定义了完整的 Props 接口，包括参数类型、默认值，让 IDE 能够提示和检查。

2️⃣ Props 可配置性

地图中心点、缩放级别、瓦片URL都可以通过 props 传入，实现组件复用。

3️⃣ 生命周期清理

onUnmounted 中调用 map.remove()，防止内存泄漏。

4️⃣ 性能优化

使用 shallowRef 而非 ref，避免 Leaflet 实例的深度响应式。

5️⃣ 窗口自适应

使用 setTimeout + invalidateSize() 确保地图尺寸正确。
